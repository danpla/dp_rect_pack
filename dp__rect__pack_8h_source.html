68 // A note on the implementation.
69 // The current algorithm is absolutely the same as in version 1.0.0,
70 // except that we only keep the leaf nodes of the binary tree. This
71 // dramatically improves performance and reduces memory usage, but
72 // growDown() and growRight() methods are harder to understand
73 // because the leafs insertion order depends on several layers of
74 // parent branches that don't physically exist. I added comments to
75 // help you visualize what happens, but it will probably be easier

338 template<typename GeomT>
339 typename RectPacker<GeomT>::InsertResult
340 RectPacker<GeomT>::insert(GeomT width, GeomT height)
341 {
342  InsertResult result;
343 
344  if (width < 0 || height < 0) {
345  result.status = InsertStatus::negativeSize;
346  return result;
347  }
348 
349  if (width == 0 || height == 0) {
350  result.status = InsertStatus::zeroSize;
351  return result;
352  }
353 
354  if (width > ctx.maxSize.w || height > ctx.maxSize.h) {
355  result.status = InsertStatus::rectTooBig;
356  return result;
357  }
358 
359  const Size rect(width, height);
360 
361  for (std::size_t i = 0, size = pages.size(); i < size; ++i)
362  if (pages[i].insert(ctx, rect, result.pos)) {
363  result.status = InsertStatus::ok;
364  result.pageIndex = i;
365  return result;
366  }
367 
368  pages.push_back(Page());
369  Page& page = pages.back();
370  page.insert(ctx, rect, result.pos);
371  result.status = InsertStatus::ok;
372  result.pageIndex = pages.size() - 1;
373 
374  return result;
375 }

378 template<typename GeomT>
379 bool RectPacker<GeomT>::Page::insert(
380  Context& ctx, const Size& rect, Position& pos)
381 {
382  assert(rect.w > 0);
383  assert(rect.w <= ctx.maxSize.w);
384  assert(rect.h > 0);
385  assert(rect.h <= ctx.maxSize.h);
386 
387  // The first insertion should be handled especially since
388  // growRight() and growDown() add spacing between the root
389  // and the inserted rectangle.
390  if (rootSize.w == 0) {
391  rootSize = rect;
392  pos.x = ctx.padding.left;
393  pos.y = ctx.padding.top;
394 
395  return true;
396  }
397 
398  return tryInsert(ctx, rect, pos) || tryGrow(ctx, rect, pos);
399 }

446 template<typename GeomT>
447 void RectPacker<GeomT>::Page::subdivideNode(
448  Context& ctx, std::size_t nodeIdx, const Size& rect)
449 {
450  assert(nodeIdx < nodes.size());
451 
452  Node& node = nodes[nodeIdx];
453 
454  assert(node.size.w >= rect.w);
455  const GeomT rightW = node.size.w - rect.w;
456  const bool hasSpaceRight = rightW > ctx.spacing.x;
457  assert(node.size.h >= rect.h);
458  const GeomT bottomH = node.size.h - rect.h;
459  const bool hasSpaceBelow = bottomH > ctx.spacing.y;
460 
461  if (hasSpaceRight) {
462  // Right node replaces the current
463 
464  const GeomT bottomX = node.pos.x;
465  const GeomT bottomW = node.size.w;
466 
467  node.pos.x += rect.w + ctx.spacing.x;
468  node.size.w = rightW - ctx.spacing.x;
469  node.size.h = rect.h;
470 
471  if (hasSpaceBelow) {
472  nodes.insert(
473  nodes.begin() + nodeIdx + 1,
474  Node(
475  bottomX,
476  node.pos.y + rect.h + ctx.spacing.y,
477  bottomW,
478  bottomH - ctx.spacing.y));
479 
480  if (nodeIdx <= growDownRootBootomIdx)
481  ++growDownRootBootomIdx;
482  }
483  } else if (hasSpaceBelow) {
484  // Bottom node replaces the current
485  node.pos.y += rect.h + ctx.spacing.y;
486  node.size.h = bottomH - ctx.spacing.y;
487  } else {
488  nodes.erase(nodes.begin() + nodeIdx);
489  if (nodeIdx < growDownRootBootomIdx)
490  --growDownRootBootomIdx;
491  }
492 }

495 template<typename GeomT>
496 bool RectPacker<GeomT>::Page::tryGrow(
497  Context& ctx, const Size& rect, Position& pos)
498 {
499  assert(ctx.maxSize.w >= rootSize.w);
500  const GeomT freeW = ctx.maxSize.w - rootSize.w;
501  assert(ctx.maxSize.h >= rootSize.h);
502  const GeomT freeH = ctx.maxSize.h - rootSize.h;
503 
504  const bool canGrowDown = (
505  freeH >= rect.h && freeH - rect.h >= ctx.spacing.y);
506  const bool mustGrowDown = (
507  canGrowDown
508  && freeW >= ctx.spacing.x
509  && (rootSize.w + ctx.spacing.x
510  >= rootSize.h + rect.h + ctx.spacing.y));
511  if (mustGrowDown) {
512  growDown(ctx, rect, pos);
513  return true;
514  }
515 
516  const bool canGrowRight = (
517  freeW >= rect.w && freeW - rect.w >= ctx.spacing.x);
518  if (canGrowRight) {
519  growRight(ctx, rect, pos);
520  return true;
521  }
522 
523  if (canGrowDown) {
524  growDown(ctx, rect, pos);
525  return true;
526  }
527 
528  return false;
529 }

532 template<typename GeomT>
533 void RectPacker<GeomT>::Page::growDown(
534  Context& ctx, const Size& rect, Position& pos)
535 {
536  assert(ctx.maxSize.h > rootSize.h);
537  assert(ctx.maxSize.h - rootSize.h >= rect.h);
538  assert(ctx.maxSize.h - rootSize.h - rect.h >= ctx.spacing.y);
539 
540  pos.x = ctx.padding.left;
541  pos.y = ctx.padding.top + rootSize.h + ctx.spacing.y;
542 
543  if (rootSize.w < rect.w) {
544  if (rect.w - rootSize.w > ctx.spacing.x) {
545  // The auxiliary node becomes the right child of the new
546  // root. It contains the current root (bottom child) and
547  // free space at the current root's right (right child).
548  nodes.insert(
549  nodes.begin(),
550  Node(
551  ctx.padding.left + rootSize.w + ctx.spacing.x,
552  ctx.padding.top,
553  rect.w - rootSize.w - ctx.spacing.x,
554  rootSize.h));
555  ++growDownRootBootomIdx;
556  }
557 
558  rootSize.w = rect.w;
559  } else if (rootSize.w - rect.w > ctx.spacing.x) {
560  // Free space at the right of the inserted rect becomes the
561  // right child of the rect's node, which in turn is the
562  // bottom child of the new root.
563  nodes.insert(
564  nodes.begin() + growDownRootBootomIdx,
565  Node(
566  pos.x + rect.w + ctx.spacing.x,
567  pos.y,
568  rootSize.w - rect.w - ctx.spacing.x,
569  rect.h));
570 
571  // The inserted node is visited before the node from the next
572  // growDown() since the current new root will be the right
573  // child of the next root.
574  ++growDownRootBootomIdx;
575  }
576 
577  rootSize.h += ctx.spacing.y + rect.h;
578 }

581 template<typename GeomT>
582 void RectPacker<GeomT>::Page::growRight(
583  Context& ctx, const Size& rect, Position& pos)
584 {
585  assert(ctx.maxSize.w > rootSize.w);
586  assert(ctx.maxSize.w - rootSize.w >= rect.w);
587  assert(ctx.maxSize.w - rootSize.w - rect.w >= ctx.spacing.x);
588 
589  pos.x = ctx.padding.left + rootSize.w + ctx.spacing.x;
590  pos.y = ctx.padding.top;
591 
592  if (rootSize.h < rect.h) {
593  if (rect.h - rootSize.h > ctx.spacing.y)
594  // The auxiliary node becomes the bottom child of the
595  // new root. It contains the current root (right child)
596  // and free space at the current root's bottom, if any
597  // (bottom child).
598  nodes.insert(
599  nodes.end(),
600  Node(
601  ctx.padding.left,
602  ctx.padding.top + rootSize.h + ctx.spacing.y,
603  rootSize.w,
604  rect.h - rootSize.h - ctx.spacing.y));
605 
606  rootSize.h = rect.h;
607  } else if (rootSize.h - rect.h > ctx.spacing.y) {
608  // Free space at the bottom of the inserted rect becomes the
609  // bottom child of the rect's node, which in turn is the
610  // right child of the new root node.
611  nodes.insert(
612  nodes.begin(),
613  Node(
614  pos.x,
615  pos.y + rect.h + ctx.spacing.y,
616  rect.w,
617  rootSize.h - rect.h - ctx.spacing.y));
618  ++growDownRootBootomIdx;
619  }
620 
621  rootSize.w += ctx.spacing.x + rect.w;
622 }
