332 template<typename GeomT, typename IndexT>
333 typename RectPacker<GeomT, IndexT>::InsertResult
334 RectPacker<GeomT, IndexT>::insert(GeomT width, GeomT height)
335 {
336  InsertResult result;
337 
338  if (width < 0 || height < 0) {
339  result.status = InsertStatus::negativeSize;
340  return result;
341  }
342 
343  if (width == 0 || height == 0) {
344  result.status = InsertStatus::zeroSize;
345  return result;
346  }
347 
348  if (width > ctx.maxSize.w || height > ctx.maxSize.h) {
349  result.status = InsertStatus::rectTooBig;
350  return result;
351  }
352 
353  const Size rect(width, height);
354 
355  for (std::size_t i = 0, size = pages.size(); i < size; ++i)
356  if (pages[i].insert(ctx, rect, result.pos)) {
357  result.status = InsertStatus::ok;
358  result.pageIndex = i;
359  return result;
360  }
361 
362  pages.push_back(Page());
363  Page& page = pages.back();
364  page.insert(ctx, rect, result.pos);
365  result.status = InsertStatus::ok;
366  result.pageIndex = pages.size() - 1;
367 
368  return result;
369 }

372 template<typename GeomT, typename IndexT>
373 bool RectPacker<GeomT, IndexT>::Page::insert(
374  Context& ctx, const Size& rect, Position& pos)
375 {
376  assert(rect.w > 0);
377  assert(rect.w <= ctx.maxSize.w);
378  assert(rect.h > 0);
379  assert(rect.h <= ctx.maxSize.h);
380 
381  Node& root = nodes[0];
382  if (root.size.w == 0) {
383  assert(nodes.size() == 1);
384 
385  root.size = rect;
386  root.rightIdx = 0;
387  root.bottomIdx = 0;
388 
389  pos.x = ctx.padding.left;
390  pos.y = ctx.padding.top;
391  return true;
392  }
393 
394  return tryInsert(ctx, rect, pos) || tryGrow(ctx, rect, pos);
395 }

412 template<typename GeomT, typename IndexT>
413 bool RectPacker<GeomT, IndexT>::Page::findNode(
414  Context& ctx, const Size& rect, IndexT& nodeIdx, Position& pos) const
415 {
416  assert(ctx.stack.empty());
417 
418  pos.x = ctx.padding.left;
419  pos.y = ctx.padding.top;
420  nodeIdx = rootIdx;
421 
422  while (true) {
423  const Node& node = nodes[nodeIdx];
424 
425  if (rect.w <= node.size.w && rect.h <= node.size.h) {
426  if (node.isEmpty()) {
427  ctx.stack.clear();
428  return true;
429  } else if (node.rightIdx != 0) {
430  if (node.bottomIdx != 0) {
431  StackState state = {
432  node.bottomIdx,
433  {
434  pos.x,
435  GeomT(
436  pos.y + node.size.h
437  - nodes[node.bottomIdx].size.h)
438  }
439  };
440  ctx.stack.push_back(state);
441  }
442 
443  nodeIdx = node.rightIdx;
444  pos.x += node.size.w - nodes[node.rightIdx].size.w;
445  continue;
446  } else if (node.bottomIdx != 0) {
447  nodeIdx = node.bottomIdx;
448  pos.y += node.size.h - nodes[node.bottomIdx].size.h;
449  continue;
450  }
451  }
452 
453  if (ctx.stack.empty())
454  return false;
455  else {
456  const StackState& state = ctx.stack.back();
457  nodeIdx = state.nodeIdx;
458  pos = state.pos;
459  ctx.stack.pop_back();
460  }
461  }

479 template<typename GeomT, typename IndexT>
480 void RectPacker<GeomT, IndexT>::Page::subdivideNode(
481  Context& ctx, IndexT nodeIdx, const Size& rect)
482 {
483  assert(nodeIdx < nodes.size());
484  assert(nodes[nodeIdx].isEmpty());
485 
486  // We insert the nodes in the same order findNode() will visit them.
487 
488  {
489  Node& node = nodes[nodeIdx];
490  assert(node.size.w >= rect.w);
491  const GeomT rightW = node.size.w - rect.w;
492  if (rightW > ctx.spacing.x) {
493  node.rightIdx = nodes.size();
494  nodes.push_back(Node(rightW - ctx.spacing.x, rect.h));
495  } else
496  node.rightIdx = 0;
497  }
498 
499  {
500  Node& node = nodes[nodeIdx];
501  assert(node.size.h >= rect.h);
502  const GeomT bottomH = node.size.h - rect.h;
503  if (bottomH > ctx.spacing.y) {
504  node.bottomIdx = nodes.size();
505  nodes.push_back(Node(node.size.w, bottomH - ctx.spacing.y));
506  } else
507  node.bottomIdx = 0;
508  }
509 }

512 template<typename GeomT, typename IndexT>
513 bool RectPacker<GeomT, IndexT>::Page::tryGrow(
514  Context& ctx, const Size& rect, Position& pos)
515 {
516  assert(rootIdx < nodes.size());
517  const Node& root = nodes[rootIdx];
518 
519  assert(ctx.maxSize.w >= root.size.w);
520  const GeomT freeW = ctx.maxSize.w - root.size.w;
521  assert(ctx.maxSize.h >= root.size.h);
522  const GeomT freeH = ctx.maxSize.h - root.size.h;
523 
524  const bool canGrowDown = (
525  freeH >= rect.h && freeH - rect.h >= ctx.spacing.y);
526  const bool mustGrowDown = (
527  canGrowDown
528  && freeW >= ctx.spacing.x
529  && (root.size.w + ctx.spacing.x
530  >= root.size.h + rect.h + ctx.spacing.y));
531  if (mustGrowDown) {
532  pos.x = ctx.padding.left;
533  pos.y = ctx.padding.top + root.size.h + ctx.spacing.y;
534  growDown(ctx, rect);
535  return true;
536  }
537 
538  const bool canGrowRight = (
539  freeW >= rect.w && freeW - rect.w >= ctx.spacing.x);
540  if (canGrowRight) {
541  pos.x = ctx.padding.left + root.size.w + ctx.spacing.x;
542  pos.y = ctx.padding.top;
543  growRight(ctx, rect);
544  return true;
545  }
546 
547  return false;
548 }

551 template<typename GeomT, typename IndexT>
552 void RectPacker<GeomT, IndexT>::Page::growDown(Context& ctx, const Size& rect)
553 {
554  std::size_t nextIdx = nodes.size();
555  const std::size_t newRootIdx = nextIdx++;
556 
557  assert(rootIdx < nodes.size());
558  const Size rootSize = nodes[rootIdx].size;
559  assert(ctx.maxSize.h > rootSize.h);
560  assert(ctx.maxSize.h - rootSize.h >= rect.h);
561  assert(ctx.maxSize.h - rootSize.h - rect.h >= ctx.spacing.y);
562 
563  const GeomT newRootW = std::max(rootSize.w, rect.w);
564  nodes.push_back(
565  Node(newRootW, rootSize.h + rect.h + ctx.spacing.y, rootIdx, 0));
566 
567  if (rootSize.w < newRootW && newRootW - rootSize.w > ctx.spacing.x) {
568  nodes.back().rightIdx = nextIdx++;
569  nodes.push_back(Node(newRootW, rootSize.h, nextIdx++, rootIdx));
570  nodes.push_back(
571  Node(newRootW - rootSize.w - ctx.spacing.x, rootSize.h));
572  }
573 
574  nodes[newRootIdx].bottomIdx = nextIdx++;
575  nodes.push_back(Node(newRootW, rect.h, 0, 0));
576 
577  if (rect.w < newRootW && newRootW - rect.w > ctx.spacing.x) {
578  nodes.back().rightIdx = nextIdx++;
579  nodes.push_back(Node(newRootW - rect.w - ctx.spacing.x, rect.h));
580  }
581 
582  rootIdx = newRootIdx;
583 }

586 template<typename GeomT, typename IndexT>
587 void RectPacker<GeomT, IndexT>::Page::growRight(Context& ctx, const Size& rect)
588 {
589  std::size_t nextIdx = nodes.size();
590  const std::size_t newRootIdx = nextIdx++;
591 
592  assert(rootIdx < nodes.size());
593  const Size rootSize = nodes[rootIdx].size;
594  assert(ctx.maxSize.w > rootSize.w);
595  assert(ctx.maxSize.w - rootSize.w >= rect.w);
596  assert(ctx.maxSize.w - rootSize.w - rect.w >= ctx.spacing.x);
597 
598  const GeomT newRootH = std::max(rootSize.h, rect.h);
599  nodes.push_back(
600  Node(rootSize.w + rect.w + ctx.spacing.x, newRootH, 0, rootIdx));
601 
602  if (rootSize.h < newRootH && newRootH - rootSize.h > ctx.spacing.y) {
603  nodes.back().bottomIdx = nextIdx++;
604  nodes.push_back(Node(rootSize.w, newRootH, rootIdx, nextIdx++));
605  nodes.push_back(
606  Node(rootSize.w, newRootH - rootSize.h - ctx.spacing.y));
607  }
608 
609  nodes[newRootIdx].rightIdx = nextIdx++;
610  nodes.push_back(Node(rect.w, newRootH, 0, 0));
611 
612  if (rect.h < newRootH && newRootH - rect.h > ctx.spacing.y) {
613  nodes.back().bottomIdx = nextIdx++;
614  nodes.push_back(Node(rect.w, newRootH - rect.h - ctx.spacing.y));
615  }
616 
617  rootIdx = newRootIdx;
618 }
